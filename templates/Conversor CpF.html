print('Conversor de Celsius/Fahrenheit')
# Mostramos o nome do programa para o usuário.

modo = 0
# Criamos uma variável chamada "modo", que define como o programa vai operar.

while modo != 3:
modo = int(input('Digite 1 para converter de ºC a ºF, digite 2 para converter de ºF a °C e 3 para sair: '))
# Pedimos para o usuário escolher um modo de conversão, usando "input". "int" especifica que é um número inteiro.
# Para que o número 3 feche o programa, conforme as instruções, precisamos criar um loop que rode indefinidamente, até o usuário
# digitar 3. Por isso, utilizamos "while" para detectar se o modo é diferente de 3.

if modo < 3:
temperatura = float(input('Qual a temperatura? '))
# Se o usuário escolheu 1 ou 2, pedimos que ele digite a temperatura que será convertida.
# Usamos float para permitir operações numéricas com casas decimais.

if modo == 1:
resultado = (temperatura * 9 / 5 + 32)
# Fórmula para converter de Celsius para Fahrenheit: (ºC × 9/5) + 32
print(f'{temperatura}ºC equivale a {resultado}°F')
# Mostramos o resultado da conversão.

elif modo == 2:
resultado = ((temperatura - 32) * 5 / 9)
# Fórmula para converter de Fahrenheit para Celsius: (ºF − 32) × 5/9
print(f'{temperatura}ºF equivale a {resultado}°C')
# Mostramos o resultado da conversão.

# Quando o usuário digita 3, o loop termina e o programa finaliza com a mensagem abaixo:
print('Tchau!')
