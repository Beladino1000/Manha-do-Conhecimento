    # O fatorial(!) de um número natural é o produto de todos os números até chegar nele mesmo. Por exemplo, 5! = 5x4x3x2x1 = 120


    número = input("Digite um número para calcular seu fatorial: ")
    # Pede que o usuário digite um número, e cria uma variável "número" com esse valor.

    while número.isdigit() == False:
    número = input("Valor inválido. Digite um número natural: ")
    # A função .isdigit() detecta se o valor da variável "número" (o valor que o usuário digitou) é realmente um número natural.
    # Enquanto não for um número, a função while vai continuar pedindo ao usuário que digite um número.

    número = int(número)
    # Agora que sabemos que o valor digitado é um número, precisamos especificar para o python que ele é um int (número inteiro).

    total = 1
    # Criamos uma outra variável "total", que vai ser o resultado do nosso fatorial.
    n = número
    # Criamos também uma outra variável "n", igual ao "número", que vai ser útil nas operações.


    # Como vamos precisar fazer várias multiplicação, usamos a função while.
    while n > 0:
    total = total * n
    # Enquanto o "n" for maior do que zero, multiplicamos o total pelo valor do "n". Isso acontece várias vezes.

    n -= 1
    # Essa linha subtrai 1 do valor do "n".
    # Dessa forma, a cada ciclo da função while, o "n" fica menor. Mas, quando chegar em zero, a função para de rodar.
    # Por exemplo, se o valor digitado for 5, então o "n" vai ser 5 e depois 4, 3, 2, 1.


    print(f"O fatorial de {número} é {total}")
    # Agora, só precisamos que o programa mostre uma mensagem com o nosso "total"!
    # Repare que, como fizemos as operações com a variável "n", o nosso número continua o mesmo.

    # Obs.: Se o número for zero, a função while não vai nunca rodar, mas como o total começa sendo 1, o programa vai mostrar que 0! = 1
