Código:

from random import randint
# Dentro do python, existem bibliotecas com funcionalidades extras.
# Nesse caso, importamos a função randint, dentro da biblioteca "random", que serve para gerar números aleatórios.

alvo = randint(1, 50)
# Geramos um número aleatório entre 1 e 50 usando randint().
# Esse número será o "alvo", ou seja, o número que o jogador precisa adivinhar.

número = int(input('Tente adivinhar o número em que estou pensando, de 1 a 50. Digite um número: '))
# Pedimos que o jogador chute um número. O valor digitado é convertido para inteiro e salvo em "número".

tentativas = 5
# Definimos que o jogador terá 5 tentativas para acertar o número.

while número != alvo:
# Enquanto o número digitado for diferente do alvo, queremos que o programa continue perguntando. Para isso, usamos um loop "while".

if número > alvo:
print('O número inserido é maior do que o alvo.')
else:
print('O número inserido é menor do que o alvo.')
# Damos uma dica ao usuário, falando se o número é maior ou menor que o alvo.

tentativas -= 1
if tentativas < 1:
break
# Tiramos uma tentativa a cada erro.
# Se acabarem as tentativas, a função "break" encerra automaticamente o loop.

número = int(input('Tente adivinhar o número em que estou pensando, de 1 a 50. Digite um número: '))
# Se ainda houverem tentativas, pedimos um novo palpite.

if número == alvo:
print('O número é este, parabéns!')
else:
print(f'O número era: {alvo}')
# Se o número digitado for igual ao alvo, mostramos uma mensagem de parabéns!
# Se o jogador não acertar dentro das 5 tentativas, mostramos qual era o número correto.
